{

void powerMeterSetup(void)
{
    isPowerMeterEnabled = true;
    if (!ads48.begin())
    {
        Report.println("Failed to initialize ADS at 48.");
        isPowerMeterEnabled = false;
    }
    if (!ads49.begin(0x49))
    {
        Report.println("Failed to initialize ADS at 49.");
        isPowerMeterEnabled = false;
    }
    if (!isPowerMeterEnabled)
        return;

    pinMode(ALERT0_PIN, INPUT_PULLUP);
    pinMode(ALERT1_PIN, INPUT_PULLUP);
    attachInterrupt(digitalPinToInterrupt(ALERT0_PIN), alert0, FALLING);
    attachInterrupt(digitalPinToInterrupt(ALERT1_PIN), alert1, FALLING);
    ads48.setGain(GAIN_TWO); // 2x gain   +/- 2.048V  1 bit = 1mV      0.0625mV
    ads49.setGain(GAIN_TWO); // 2x gain   +/- 2.048V  1 bit = 1mV      0.0625mV
    ads48.setDataRate(RATE_ADS1115_860SPS);
    ads49.setDataRate(RATE_ADS1115_860SPS);
    // ads48.setConversionReadyPinMode();
    // ads49.setConversionReadyPinMode();
    ads48.startADCReading(ADS1X15_REG_CONFIG_MUX_DIFF_1_3, true);
    ads49.startADCReading(ADS1X15_REG_CONFIG_MUX_DIFF_1_3, true);
    initDSP();
    Report.printf("ADS48 DataRate: %s, Gain: %s\n", rateToString(ads48.getDataRate()), gainToString(ads48.getGain()));
    Report.printf("ADS49 DataRate: %s, Gain: %s\n", rateToString(ads49.getDataRate()), gainToString(ads49.getGain()));
}

static float window[numSamples];

AnalysisResult processSamples()
{
    AnalysisResult result;
    result.mean = removeDCOffset();
    result.amplitude = goertzel();
    result.dsp = calcDFT();
    return result;
}

void initDSP()
{

    initGoertzel(60.0f); // 60Hz
    // Initialize the Blackman window
    dsps_wind_blackman_f32(window, numSamples);

    esp_err_t err = dsps_fft2r_init_fc32(nullptr, numSamples);
    if (err != ESP_OK)
    {
        Report.printf("FFT init failed: %d\n", err);
        while (true)
        {
            delay(10);
        }
    }
}

// Goertzel state
static float coeff, sine, cosine;

// Initialize Goertzel for a target frequency
void initGoertzel(float targetFreq)
{
    float normalized = 2.0f * M_PI * targetFreq / 600.0f; // normalize to sample rate
    cosine = cosf(normalized);
    sine = sinf(normalized);
    coeff = 2.0f * cosine;
}

// run Goertzel on your buffer of raw ADC readings (convert to volts if you like)
float goertzel()
{
    float Q0 = 0, Q1 = 0, Q2 = 0;
    // process each sample
    for (int i = 0; i < numSamples; i++)
    {
        // convert raw→voltage if desired:
        // float x = (samples[i] & 0x0FFF) * (3.9f / 4095.0f);
        float x = workspace[i];
        Q0 = coeff * Q1 - Q2 + x;
        Q2 = Q1;
        Q1 = Q0;
    }
    // pull out real & imag parts
    float real = Q1 - Q2 * cosine;
    float imag = Q2 * sine;
    // magnitude of the bin
    float magnitude = sqrt(real * real + imag * imag);
    // optional: scale by 2/N to get the peak amplitude of the sine wave
    return (2.0 * magnitude) / numSamples;
}

float removeDCOffset()
{
    // remove DC offset for the given channel
    for (int i = 0; i < numSamples; i++)
        workspace[i] = multiplier * sample[i]; // copy samples to workspace float array

    // remove DC offset
    float m = 0;
    for (int i = 0; i < numSamples; i++)
        m += sample[i];
    m /= (float)numSamples;
    m *= multiplier;
    for (int i = 0; i < numSamples; i++)
        workspace[i] -= m;
    return m;
}

static float fftBuf[2 * numSamples];

DFTResult calcDFT(void)
{
    // 2) Build the interleaved complex array
    for (int i = 0; i < numSamples; i++)
    {
        fftBuf[2 * i] = window[i] * workspace[i]; // real part
        fftBuf[2 * i + 1] = 0.0f;                 // imag part
    }

    // 3) Run the radix-2 complex FFT
    dsps_fft2r_fc32(fftBuf, numSamples);

    // 4) Bit-reverse reorder
    dsps_bit_rev_fc32(fftBuf, numSamples);

    DFTResult result;
    // 5) Compute magnitudes into result.dft[] array
    for (int k = 0; k < numSamples / 2; k++)
    {
        float real = fftBuf[2 * k];
        float imag = fftBuf[2 * k + 1];
        result.dft[k] = 10.0f * log10f(real * real + imag * imag + 1e-12f);
    }

    result.maxValue = result.dft[0];
    result.bestBin = 0;
    result.bestFreq = 0.0f;

    for (int i = (numSamples / 4) - 10; i < (numSamples / 4) + 10; i++)
    {
        if (result.maxValue < result.dft[i])
        {
            result.maxValue = result.dft[i];
            result.bestBin = i;
            result.bestFreq = i * (float)600 / (float)numSamples; // convert index to frequency
        }
    }
    return result;
}
